# The distribution is not symmetric , this may lead to wrong prediction of our model , so remove them.

# remove any rows without user reviews
games = games[games["users_rated"] > 0]

# remove any rows with missing values
games = games.dropna(axis = 0)

# get all the columns from the dataset
columns = games.columns.tolist()

# filter the columns to remove unwanted data
columns = [c for c in columns if c not in ["bayes_average_rating" , "average_rating" , "type" , "name" , "id"]]

# storing the variable we will be predicting on
target = "average_rating"

Train & Test Split¶

In [18]:
# generayte train & test dataset
from sklearn.model_selection import train_test_split

# generate training set
train = games.sample(frac=0.8 , random_state=1)

# select anything not in training set and put it in test set
test = games.loc[~games.index.isin(train.index)]

# print training dataset & testing dataset shapes
print(train.shape)
print(test.shape)

from sklearn.metrics import mean_squared_error
